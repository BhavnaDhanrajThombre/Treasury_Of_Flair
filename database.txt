CREATE TABLE users (

         name VARCHAR(50) NOT NULL,

         password VARCHAR(255) NOT NULL,

Insert the Items in the Database

CREATE TABLE artworks (
    id INT AUTO_INCREMENT PRIMARY KEY,
    category VARCHAR(100) NOT NULL,    -- Art category (e.g., 'Pop Art', 'Sculpture')
    title VARCHAR(255) NOT NULL,       -- Title of the artwork
    artist VARCHAR(255),               -- Artist's name (can be NULL if unknown)
    description TEXT,                  -- Full description of the piece
    image_path VARCHAR(500),           -- File path for the image
    price VARCHAR(100),                -- Price or price range
    availability VARCHAR(50),          -- Stock status or availability badge
    popularity INT DEFAULT 0,          -- Relative popularity score
    INDEX idx_category (category)
);
